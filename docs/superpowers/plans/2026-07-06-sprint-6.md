> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship Sprint 6: member balance lookup, A1/A2/A3 alerts, member statement generation/sharing, president/member public PDF delivery, and year-end two-pool share-out draft, override, approval, payouts, snapshots, and PDFs.

**Architecture:** Keep tenant mutations in domain services under `packages/domain/src`, with App Router pages/actions as thin authenticated adapters. PDF artifacts are generated from canonical JSON payloads, stored in Vercel Blob as private blobs, exposed through unlisted hash routes for WhatsApp recipients, and indexed in `statement_archive`. Year-end share-out is a draft-first workflow: compute and audit the draft, permit auditable overrides, then approve in one transaction that writes payouts, snapshots, and archives.

**Tech Stack:** Next.js App Router, React Server Components, Server Actions, Drizzle ORM, Neon/Postgres materialized views, Vitest, `@react-pdf/renderer`, Vercel Blob, Tailwind/design tokens from `@mi-banquito/ui`.

```sql
DROP MATERIALIZED VIEW IF EXISTS mv_member_compliance_state;
CREATE MATERIALIZED VIEW mv_member_compliance_state AS
SELECT
  m.org_id,
  m.id AS member_id,
  m.display_name,
  CASE
    WHEN COALESCE(SUM(c.amount), 0) >= 0 THEN 'al_dia'
    ELSE 'atrasado'
  END AS state,
  (
    COALESCE(m.initial_savings_balance, 0)
    + COALESCE(SUM(c.amount), 0)
    - COALESCE(SUM(w.amount), 0)
  )::numeric(18,4) AS current_balance,
  now() AS refreshed_at
FROM member m
LEFT JOIN contribution c
  ON c.org_id = m.org_id
 AND c.member_id = m.id
LEFT JOIN withdrawal w
  ON w.org_id = m.org_id
 AND w.member_id = m.id
WHERE m.status = 'activo'
GROUP BY m.org_id, m.id, m.display_name, m.initial_savings_balance;

CREATE UNIQUE INDEX IF NOT EXISTS idx_mv_member_compliance_state_member_id
ON mv_member_compliance_state(org_id, member_id);

CREATE INDEX IF NOT EXISTS idx_mv_member_compliance_state_display_name
ON mv_member_compliance_state(org_id, lower(display_name));

DROP MATERIALIZED VIEW IF EXISTS mv_member_time_weighted_balance;
CREATE MATERIALIZED VIEW mv_member_time_weighted_balance AS
SELECT
  m.org_id,
  m.id AS member_id,
  EXTRACT(YEAR FROM now())::integer AS fiscal_year,
  COALESCE(m.initial_savings_balance, 0)::numeric(18,4) AS accumulated_savings,
  (COALESCE(m.initial_savings_balance, 0) * 365)::numeric(18,4) AS saldo_ponderado_usd_dias,
  now() AS refreshed_at
FROM member m
WHERE m.status = 'activo';

CREATE UNIQUE INDEX IF NOT EXISTS idx_mv_member_time_weighted_balance_member_year
ON mv_member_time_weighted_balance(org_id, member_id, fiscal_year);

DROP MATERIALIZED VIEW IF EXISTS mv_loan_activity_points;
CREATE MATERIALIZED VIEW mv_loan_activity_points AS
SELECT
  l.org_id,
  COALESCE(l.member_id, lg.member_id) AS member_id,
  EXTRACT(YEAR FROM COALESCE(r.dated_on, l.originated_on))::integer AS fiscal_year,
  COALESCE(SUM(r.applied_to_principal), 0)::numeric(18,4) AS loan_activity_basis,
  now() AS refreshed_at
FROM loan l
LEFT JOIN repayment r
  ON r.org_id = l.org_id
 AND r.loan_id = l.id
LEFT JOIN loan_guarantor lg
  ON lg.org_id = l.org_id
 AND lg.loan_id = l.id
GROUP BY l.org_id, COALESCE(l.member_id, lg.member_id), EXTRACT(YEAR FROM COALESCE(r.dated_on, l.originated_on));

CREATE UNIQUE INDEX IF NOT EXISTS idx_mv_loan_activity_points_member_year
ON mv_loan_activity_points(org_id, member_id, fiscal_year);

DROP MATERIALIZED VIEW IF EXISTS mv_distributable_surplus;
CREATE MATERIALIZED VIEW mv_distributable_surplus AS
SELECT
  o.id AS org_id,
  EXTRACT(YEAR FROM now())::integer AS fiscal_year,
  COALESCE(SUM(ia.interest_amount), 0)::numeric(18,4) AS interest_total,
  COALESCE(SUM(lf.amount), 0)::numeric(18,4) AS fees_total,
  0::numeric(18,4) AS cxc_anterior,
  (COALESCE(SUM(ia.interest_amount), 0) + COALESCE(SUM(lf.amount), 0))::numeric(18,4) AS distributable_surplus,
  now() AS refreshed_at
FROM organization o
LEFT JOIN interest_accrual ia
  ON ia.org_id = o.id
LEFT JOIN loan_fee lf
  ON lf.org_id = o.id
GROUP BY o.id;

CREATE UNIQUE INDEX IF NOT EXISTS idx_mv_distributable_surplus_org_year
ON mv_distributable_surplus(org_id, fiscal_year);

CREATE UNIQUE INDEX IF NOT EXISTS uq_statement_archive_org_kind_member_period
ON statement_archive(org_id, kind, member_id, period_label)
WHERE member_id IS NOT NULL;
```

Add this method inside `createAlertsService()`:

```ts
async emitSprint6DailyAlerts(input) {
  const today = dayStartUtc(input.today);
  const dedupWindowEnd = new Date(today.getTime() + MS_PER_DAY);
  const summary: Sprint6AlertRunSummary = {
    pendingReconciliationAlertsEmitted: 0,
    loanDueSoonAlertsEmitted: 0,
    contributionLateAlertsEmitted: 0,
    skippedExisting: 0,
    failures: [],
  };

  const orgs = await db.select({ id: organization.id }).from(organization).where(eq(organization.status, "active"));
  for (const org of orgs) {
    try {
      await withTenantTransaction(org.id, async (tx) => {
        if (today.getUTCDate() === 5) {
          const prevMonth = new Intl.DateTimeFormat("es-EC", { month: "long", year: "numeric", timeZone: "UTC" })
            .format(new Date(Date.UTC(today.getUTCFullYear(), today.getUTCMonth() - 1, 1)));
          const payload = pendingReconciliationAlertPayload({ prevMonth });
          await tx.insert(alert).values({
            orgId: org.id,
            alertKind: "A1",
            severity: "high",
            audience: "treasurer",
            subjectKind: "contribution_cycle",
            subjectId: org.id,
            payload,
            dedupWindowEnd,
            dismissedAt: null,
            dismissedBy: null,
            snoozedUntil: null,
            createdAt: today,
          }).onConflictDoNothing();
          summary.pendingReconciliationAlertsEmitted += 1;
        }
      });
    } catch (error) {
      summary.failures.push({ orgId: org.id, message: error instanceof Error ? error.message : String(error) });
    }
  }

  return summary;
}
```

```ts
export function buildBalanceShareUrl(input: {
  whatsappNumber: string | null;
  memberName: string;
  currentBalance: string;
}): string | null {
  const digits = input.whatsappNumber?.replace(/\D/g, "") ?? "";
  if (!digits) return null;
  const message = `Hola ${input.memberName}, tu saldo actual en Mi Banquito es USD ${Number(input.currentBalance).toFixed(2)}.`;
  return `https://wa.me/${digits}?text=${encodeURIComponent(message)}`;
}
```

```ts
      "content-disposition": `inline; filename="${archive.kind}-${archive.periodLabel}.pdf"`,
      "cache-control": "private, max-age=300",
```

```ts
function cents4(value: string | number): bigint {
  return BigInt(Math.round(Number(value) * 10000));
}

function money4(value: bigint): string {
  const sign = value < 0n ? "-" : "";
  const abs = value < 0n ? -value : value;
  return `${sign}${abs / 10000n}.${String(abs % 10000n).padStart(4, "0")}`;
}

export function fiscalYearForDate(dateOnly: string, input: { startMonth: number; startDay: number }): number {
  const [year, month, day] = dateOnly.split("-").map(Number);
  const startsThisYear = month > input.startMonth || (month === input.startMonth && day >= input.startDay);
  return startsThisYear ? year : year - 1;
}

export function computeTwoPoolDraft(input: {
  repartoTotal: string;
  loanPoolPct: string;
  savingsPoolPct: string;
  members: Array<{ memberId: string; accumulatedSavings: string; saldoPonderadoUsdDias: string; loanActivityBasis: string }>;
}) {
  const total = cents4(input.repartoTotal);
  const loanPool = cents4(Number(input.repartoTotal) * Number(input.loanPoolPct));
  const savingsPool = total - loanPool;
  const loanBasisTotal = input.members.reduce((sum, row) => sum + cents4(row.loanActivityBasis), 0n);
  const weightedTotal = input.members.reduce((sum, row) => sum + cents4(row.saldoPonderadoUsdDias), 0n);
  const lines = input.members.map((row) => {
    const loanBonus = loanBasisTotal === 0n ? 0n : loanPool * cents4(row.loanActivityBasis) / loanBasisTotal;
    const savingsInterest = weightedTotal === 0n ? 0n : savingsPool * cents4(row.saldoPonderadoUsdDias) / weightedTotal;
    const draftShare = loanBonus + savingsInterest;
    return {
      memberId: row.memberId,
      accumulatedSavings: money4(cents4(row.accumulatedSavings)),
      loanActivityBasis: money4(cents4(row.loanActivityBasis)),
      loanBonusC: money4(loanBonus),
      savingsInterest: money4(savingsInterest),
      draftShareAmount: money4(draftShare),
      finalShareAmount: money4(draftShare),
    };
  });
  const totalDraft = lines.reduce((sum, row) => sum + cents4(row.draftShareAmount), 0n);
  return {
    loanPoolAmount: money4(loanPool),
    savingsPoolAmount: money4(savingsPool),
    totalDraft: money4(totalDraft),
    lines,
  };
}
```

```ts
  const reason = input.reason.trim();
  if (override !== 0n && !reason) throw new Error("override_reason_required");
  const lines = input.lines.map((line) => line.id === input.lineId
    ? { ...line, finalShareAmount: money4(override), overrideReason: reason }
    : line);
  const finalTotal = lines.reduce((sum, line) => sum + cents4(line.finalShareAmount), 0n);
  return {
    lines,
    ajusteAmount: money4(cents4(input.repartoTotal) - finalTotal),
  };
}
```
